\begin{algorithm}[H]
  \DontPrintSemicolon
  \SetAlgoLined
  \SetKwProg{Fn}{Function}{: }{}
  \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
  \Fn{$\hgsat$}{
    \Input{full guarded TGDs $\Sigma_f$ and non-full guarded TGDs $\Sigma_n$ }
    \Output{Atomic rewriting of $\Sigma_f \cup \Sigma_n$}
    \BlankLine
    $\workingSet = \vnf\bigl(\sk(\Sigma_n)\bigr)$\comment{working set}\;
    $\fullSet = \Sigma_f$\comment{full set}\;
    $\nonFullSet = \emptyset$\comment{non-full set}\;
    \While{$\workingSet\neq\emptyset$}{
      Let $\sigma$ be a rule in $\workingSet$\;
      \eIf{$\sigma$ is full}{
        $\fullSet = \fullSet\cup\{\sigma\}$\;
        $\evolvedSet = \hyperresolve(\nonFullSet, \sigma)$\;
      }{
        % \comment{an index }\;
        $\nonFullSet = \nonFullSet\cup\{\sigma\}$\;
	$\evolvedSet = \bigcup_{\rho \in \fullSet} \hyperresolve(\nonFullSet, \rho)$\;
      }
      $\workingSet = \workingSet\cup\evolvedSet\setminus\bigl(\fullSet\cup\nonFullSet\bigr)$\;
    }
    \KwRet{$\fullSet$}\;
  }\;
